function registrati(){ const email = document.getElementById("email") .value; const password = document.getElementById("password") .value; if(email === "" || password === ""){ document.getElementById("messaggio") .innerText = "Compila tutti i campi"; return; } // utenti esistenti let utenti = JSON.parse(localStorage.getItem("utenti")) || []; // controllo email già usata const esiste = utenti.find(u => u.email === email); if(esiste){ document.getElementById("messaggio") .innerText = "Email già registrata"; return; } // nuovo utente const nuovoUtente = { email: email, password: password, ruolo: "utente" }; utenti.push(nuovoUtente); localStorage.setItem( "utenti", JSON.stringify(utenti) ); document.getElementById("messaggio") .innerText = "Registrazione completata!"; }
